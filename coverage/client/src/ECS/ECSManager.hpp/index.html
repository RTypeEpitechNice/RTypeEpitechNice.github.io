#ifndef ECSMANAGER_HPP_
    #define ECSMANAGER_HPP_

    #include <memory>
    #include <unordered_map>
    #include <string>
    #include <vector>
    #include <thread>


    #include "Registry.hpp"
    #include "SystemManager.hpp"
    #include "AssetStorage.hpp"

namespace ECS {

    /**
    * This class manages the entire engine.
    * This is the class that is passed to scripts to be able to modify the game at will.
    * It is also the one managing and executing the registry (component and entity manager) as well as the systems
    * It is also responsible for json parsing as well as world management.
    *
    */
    class ECSManager {

        public:
            /**
            * Create the engine's class that manages all the worlds and systems.
            * @param name Name of the game you create (the name of the window)
            * @param isGraphic Enables the window and graphic systems (Drawable, animate, parallax, etc)
            * @param isLogic Enables the logic systems (collisions, scripts, etc)
            * @param preparedWindow shared_ptr to a window you setup that overrides the default one. It is not necessary
            *
            */
            ECSManager(const std::string &name = "Game", const bool &isGraphic = true, const bool &isLogic = true, std::shared_ptr<sf::RenderWindow> preparedWindow = nullptr);
            ~ECSManager();

            /**
            * Create an empty world with a name
            * @param path Name of the world to create
            *
            */
            void createWorld(const std::string& worldName);

            /**
            * Changes the currently active world
            * @param worldName name of the world
            *
            */
            void changeWorld(const std::string& worldName);

            /**
            * Load a world from a configuration file
            * @param path Path of the world's configuration file
            *
            */
            void loadWorld(const std::string& filePath);

            /**
            * Runs all of the systems at once
            *
            */
            void runSystems();

            /**
            * Launches the game engine's loop
            *
            */
            void run();

            /**
            * Returns the current world's registry
            * @return Current registry of the world
            *
            */
            Registry& getCurrentRegistry();

            /**
            * Create an empty entity in the current world
            *
            * @return ID of the created entity
            */
            const ECS::entity_id &createEntity();

            /**
            * Create an entity from a prefab in the AssetStorage singleton
            * @param prefabName name of the prefab
            * 
            * @return ID of the created entity
            *
            */
            ECS::entity_id createEntityFromPrefab(const std::string& prefabName);

            /**
            * Destroy an entity in the current world
            * @param id Entity ID to destroy
            *
            */
            void destroyEntity(const entity_id &id);

            /**
            * Returns the used window's pointer wrapped in a shared::ptr
            *
            * @return SFML window
            */
            std::shared_ptr<sf::RenderWindow> getWindow();

            /**
            * Returns elaped time this frame in seconds
            * @return Elapsed time
            *
            */
            const double &getDeltaTime();

            /**
            * Create a world from a world configuration
            * @tparam System Struct of the system
            * @tparam Args Arguments required by the system at initialisation (usually the window)
            *
            */
            template <typename System, typename... Args>
            void addSystem(Args&&... args)
            {
                this->_systemManager->addSystem<System>(std::forward<Args>(args)...);
            }

            /**
            * Add a component to an entity in the current world
            * @param id Entity ID
            * @tparam Component Component's struct which's values are setup or not
            *
            */
            template <typename Component>
            void addComponent(entity_id id, Component comp)
            {
                this->getCurrentRegistry().addComponent(id, comp);
            }

            /**
            * Remove a component from an entity in the current world
            * @param id Entity ID
            * @tparam Component Component's struct
            * 
            * @warning It should be called like this: removeComponent<ComponentStructType>(id)
            *
            */
            template <typename Component>
            void removeComponent(entity_id id)
            {
                this->getCurrentRegistry().removeComponent<Component>(id);
            }

            /**
            * Get a component from an entity in the current world, you can then modify the shared_ptr component
            * to directly modify it's information inside the entity, no need to re-add it
            * @param id Entity ID
            * @tparam Component Component's struct
            * 
            * @warning It should be called like this: getComponent<ComponentStructType>(id)
            * 
            * @return Entity's component wrapped in a shared_ptr
            *
            */
            template <typename Component>
            std::shared_ptr<Component> getComponent(entity_id id)
            {
                return this->getCurrentRegistry().getComponent<Component>(id);
            }

            /**
            * Returns a list of all the entities that have a component type, usefull when you create a System
            * or want to know only certain entities to increase performance
            * @tparam Component Component's struct
            * 
            * @warning It should be called like this: getEntitiesWithComponent<ComponentStructType>()
            *
            */
            template <typename Component>
            std::vector<entity_id> getEntitiesWithComponent()
            {
                return this->getCurrentRegistry().getEntitiesWithComponent<Component>();
            }

            /**
             * test
             */
            void loopServer();

        private:
            std::shared_ptr<sf::RenderWindow> _win;
            double _dt;

            std::unordered_map<std::string, std::shared_ptr<Registry>> worlds;
            std::unique_ptr<SystemManager> _systemManager;
            std::shared_ptr<Registry> _currentRegistry;

            std::unordered_map<std::string, std::function<void(nlohmann::json, ECS::entity_id)>> _jsonToComp;

            void loadEntitiesFromJson(const nlohmann::json& worldData);
            ECS::entity_id instantiatePrefab(const std::string& prefabName, const nlohmann::json& overrideData);
            void windowLoop();

            template <typename T>
            T get_json_value(const nlohmann::json& j, const std::string& key) {
                if (j.contains(key)) {
                    return j[key].get<T>();
                } else {
                    return T();
                }
            }

        };

    }

#endif
