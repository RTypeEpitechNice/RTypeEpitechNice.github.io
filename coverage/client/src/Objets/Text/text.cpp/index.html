#include "text.hpp"

namespace RType {
    namespace GUI {
        namespace Objet {
            Text::Text(std::pair<float, float> position, const std::string& text, unsigned int charSize, const sf::Color& color, const std::string& fontType) {
                _font = ResourceManager::getFont(fontType);
                _text.setFont(_font);
                _text.setString(text);
                _text.setCharacterSize(charSize);
                _text.setFillColor(color);
                _text.setPosition(position.first, position.second);
                _fontSize = charSize;
            }

            void Text::draw(sf::RenderWindow& window) const {
                window.draw(_text);
            }

            void Text::resize(float factor) {
                _text.setCharacterSize(_fontSize);
            }

            void Text::setPosition(sf::Vector2f position) {
                _text.setPosition(position);
            }

            void Text::setColor(sf::Color color) {
                _text.setFillColor(color);
            }

            void Text::setLabel(const std::string& label) {
                _text.setString(sf::String::fromUtf8(label.begin(), label.end()));
            }
        }
    }
