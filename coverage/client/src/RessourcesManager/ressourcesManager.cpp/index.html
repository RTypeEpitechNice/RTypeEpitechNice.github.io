#include "ressourcesManager.hpp"

namespace RType {
    namespace GUI {

        std::unordered_map<std::string, sf::Texture> ResourceManager::_textures;
        std::unordered_map<std::string, sf::Font> ResourceManager::_fonts;
        std::unordered_map<std::string, std::unique_ptr<sf::Music>> ResourceManager::_musics;
        std::unordered_map<std::string, sf::SoundBuffer> ResourceManager::_sounds;
        std::unordered_map<std::string, sf::Sound> ResourceManager::_activeSounds;
        bool ResourceManager::_soundEnabled = true;
        bool ResourceManager::_musicEnabled = true;

        void ResourceManager::loadAllResources() {
            loadTexture("backgroundMenu.jpg", "assets/background/backgroundMenu.jpg");
            loadTexture("backgroundSetting.jpg", "assets/background/backgroundSetting.jpg");
            loadTexture("logo.png", "assets/logo/Epitech_logo.png");
            loadTexture("Arnaud.png", "assets/credit/Arnaud.png");
            loadTexture("Clement.png", "assets/credit/Clement.jpg");
            loadTexture("Alexandre.png", "assets/credit/Alexandre.png");
            loadTexture("Jule.png", "assets/credit/Jule.png");
            loadTexture("Lucas.png", "assets/credit/Lucas.jpg");
            loadTexture("button.png", "assets/button/button.png");
            loadTexture("button2.png", "assets/button/button2.png");
            loadTexture("backButton.png", "assets/button/backButton.png");
            loadFont("Arial.ttf", "assets/fonts/arial.ttf");
            loadMusic("MenuMusic.wav", "assets/music/MenuMusic.wav");
            loadSound("buttonClick.wav", "assets/sounds/buttonClick.wav");
        }

        void ResourceManager::clearResources() {
            _textures.clear();
            _fonts.clear();
            _musics.clear();
            _sounds.clear();
            _activeSounds.clear();
        }

        //////////////////////////////////// TEXTURES ////////////////////////////////////

        bool ResourceManager::loadTexture(const std::string& textureLabel, const std::string& filename) {
            if (isTextureLoaded(textureLabel))
                return true;
            sf::Texture texture;
            if (!texture.loadFromFile(filename)) {
                std::cerr << "Failed to load texture: " << filename << std::endl;
                return false;
            }
            _textures[textureLabel] = std::move(texture);
            return true;
        }

        sf::Texture& ResourceManager::getTexture(const std::string& textureLabel) {
            auto thisTexture = _textures.find(textureLabel);
            if (thisTexture == _textures.end()) {
                throw std::runtime_error("Texture not found: " + textureLabel);
            }
            return thisTexture->second;
        }

        bool ResourceManager::isTextureLoaded(const std::string& textureLabel) {
            return _textures.find(textureLabel) != _textures.end();
        }

        //////////////////////////////////// FONTS ///////////////////////////////////////

        bool ResourceManager::loadFont(const std::string& fontLabel, const std::string& filename) {
            if (isFontLoaded(fontLabel))
                return true;
            sf::Font font;
            if (!font.loadFromFile(filename)) {
                std::cerr << "Failed to load font: " << filename << std::endl;
                return false;
            }
            _fonts[fontLabel] = std::move(font);
            return true;
        }

        sf::Font& ResourceManager::getFont(const std::string& fontLabel) {
            auto thisFont = _fonts.find(fontLabel);
            if (thisFont == _fonts.end()) {
                throw std::runtime_error("Font not found: " + fontLabel);
            }
            return thisFont->second;
        }

        bool ResourceManager::isFontLoaded(const std::string& fontLabel) {
            return _fonts.find(fontLabel) != _fonts.end();
        }

        //////////////////////////////////// MUSIC ///////////////////////////////////////

        bool ResourceManager::loadMusic(const std::string& musicLabel, const std::string& filename) {
            if (isMusicLoaded(musicLabel))
                return true;
            auto music = std::make_unique<sf::Music>();
            if (!music->openFromFile(filename)) {
                std::cerr << "Failed to load music: " << filename << std::endl;
                return false;
            }
            music->setVolume(100);
            _musics[musicLabel] = std::move(music);
            return true;
        }

        sf::Music* ResourceManager::getMusic(const std::string& musicLabel) {
            auto thisMusic = _musics.find(musicLabel);
            if (thisMusic == _musics.end()) {
                throw std::runtime_error("Music not found: " + musicLabel);
            }
            return thisMusic->second.get();
        }

        bool ResourceManager::isMusicLoaded(const std::string& musicLabel) {
            return _musics.find(musicLabel) != _musics.end();
        }

        void ResourceManager::updateMusicVolume(const std::string& musicLabel, float volume) {
            sf::Music* music = getMusic(musicLabel);
            if (music) {
                music->setVolume(volume);
            }
        }

        //////////////////////////////////// SOUNDS ///////////////////////////////////////

        bool ResourceManager::loadSound(const std::string& soundLabel, const std::string& filename) {
            if (isSoundLoaded(soundLabel))
                return true;
            sf::SoundBuffer soundBuffer;
            if (!soundBuffer.loadFromFile(filename)) {
                std::cerr << "Failed to load sound: " << filename << std::endl;
                return false;
            }
            _sounds[soundLabel] = std::move(soundBuffer);
            return true;
        }

        sf::SoundBuffer& ResourceManager::getSound(const std::string& soundLabel) {
            auto thisSound = _sounds.find(soundLabel);
            if (thisSound == _sounds.end()) {
                throw std::runtime_error("Sound not found: " + soundLabel);
            }
            return thisSound->second;
        }

        bool ResourceManager::isSoundLoaded(const std::string& soundLabel) {
            return _sounds.find(soundLabel) != _sounds.end();
        }

        void ResourceManager::playSound(const std::string& soundLabel) {
            if (!_soundEnabled)
                return;
            if (!isSoundLoaded(soundLabel))
                loadSound(soundLabel, soundLabel);
            if (_activeSounds.find(soundLabel) != _activeSounds.end()) {
                _activeSounds[soundLabel].play();
            } else {
                sf::Sound sound;
                sound.setBuffer(getSound(soundLabel));
                sound.setVolume(100.f);
                sound.play();
                _activeSounds[soundLabel] = std::move(sound);
            }
        }

        void ResourceManager::updateAllSoundsVolume(float volume) {
            for (auto& [label, sound] : _activeSounds) {
                sound.setVolume(volume);
            }
        }

        void ResourceManager::clearInactiveSounds() {
            for (auto it = _activeSounds.begin(); it != _activeSounds.end();) {
                if (it->second.getStatus() != sf::Sound::Playing) {
                    it = _activeSounds.erase(it);
                } else {
                    ++it;
                }
            }
        }
    }
