#ifndef SYSTEMS_HPP_
    #define SYSTEMS_HPP_

    #include <iostream>
    #include "GUI_SFML_Graphics.hpp"

    #include "Registry.hpp"

    namespace ECS::System
    {

        /**
        * Manages all the entities movement by adding their velocity to their position.
        *
        */
        struct Movement {

            void operator()(ECS::Registry &R, double dt)
            {
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::Velocity>();
                for (auto &entity : allEntities) {
                    auto pos = R.getComponent<ECS::Component::Position>(entity);
                    auto vel = R.getComponent<ECS::Component::Velocity>(entity);
                    if (vel != nullptr) {
                        pos->x += (vel->x * dt);
                        pos->y += (vel->y * dt);
                    }
                }
            }
        };

        /**
        * Manages the collisions between entities and execute their's scripts "onCollision()" method if they do
        *
        */
        struct Collision {

            bool checkCollision(const ECS::Component::Hitbox &a, const ECS::Component::Hitbox &b) {
                return !(a.x + a.w < b.x || a.x > b.x + b.w || a.y + a.h < b.y || a.y > b.y + b.h);
            }

            void operator()(ECS::Registry &R, double dt)
            {
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::Hitbox>();
                ECS::Component::Hitbox a;
                ECS::Component::Hitbox b;

                for (size_t i = 0; i < allEntities.size(); ++i) {
                    auto &entity = allEntities[i];
                    auto aHitbox = R.getComponent<ECS::Component::Hitbox>(entity);
                    auto aPos = R.getComponent<ECS::Component::Position>(entity);

                    if (aPos == nullptr) {
                        continue;
                    }

                    a = *aHitbox;
                    a.x += aPos->x;
                    a.y += aPos->y;

                    // Only check entities that come after the current one to avoid double-checking pairs
                    for (size_t j = i + 1; j < allEntities.size(); ++j) {
                        auto &otherEntity = allEntities[j];
                        auto bHitbox = R.getComponent<ECS::Component::Hitbox>(otherEntity);
                        auto bPos = R.getComponent<ECS::Component::Position>(otherEntity);

                        if (bPos == nullptr) {
                            continue;
                        }

                        b = *bHitbox;
                        b.x += bPos->x;
                        b.y += bPos->y;

                        if (checkCollision(a, b)) {
                            auto script_one = R.getComponent<ECS::Component::Script>(entity);
                            auto script_two = R.getComponent<ECS::Component::Script>(otherEntity);
                            if (script_one != nullptr) {
                                script_one->content->onCollision(otherEntity);
                            }
                            if (script_two != nullptr) {
                                script_two->content->onCollision(entity);
                            }
                        }
                    }
                }
            }
        };

        /**
        * Executes all active script's "onUpdate()" and "onStart()" methods.
        *
        */
        struct ScriptUpdate {

            void operator()(ECS::Registry &R, double dt) {
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::Script>();

                for (auto &entity : allEntities) {
                    auto s = R.getComponent<ECS::Component::Script>(entity);

                    if (s != nullptr) {
                        if (!s->hasStarted) {
                            s->content->onStart();
                            s->hasStarted = true;
                        }
                        s->content->onUpdate();
                    }
                }
            }
        };

        /**
        * Executes the entity's "onClick()" method when the mouse clicks on it.
        *
        * @param _win the engine's window.
        * @param mousePos Mouse position value, it is modified each frame.
        */
        struct ScriptOnClick {

            std::shared_ptr<sf::RenderWindow> _win;
            sf::Vector2i mousePos;

            bool checkCollision(std::shared_ptr<ECS::Component::Hitbox> rect, float pointX, float pointY) {
                return !(pointX < rect->x || pointX > rect->x + rect->w || pointY < rect->y || pointY > rect->y + rect->h);
            }

            ScriptOnClick( std::shared_ptr<sf::RenderWindow>& winPtr) : _win(winPtr) {}

            void operator()(ECS::Registry &R, double dt) {
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::Script>();

                mousePos = sf::Mouse::getPosition(*_win);

                for (auto &entity : allEntities) {
                    auto s = R.getComponent<ECS::Component::Script>(entity);
                    auto h = R.getComponent<ECS::Component::Hitbox>(entity);

                    if (s != nullptr && h != nullptr) {
                        if (checkCollision(h, sf::Mouse::getPosition().x, sf::Mouse::getPosition().y)
                            && sf::Mouse::isButtonPressed(sf::Mouse::Left)) {
                            s->content->onClick();
                        }
                    }
                }
            }
        };

        /**
        * Render's all active Drawable components on the window
        *
        * @param _win the engine's window.
        */
        struct Draw {

            std::shared_ptr<sf::RenderWindow> _win;

            Draw( std::shared_ptr<sf::RenderWindow>& winPtr) : _win(winPtr) {}

            void operator()(ECS::Registry &R, double dt)
            {
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::Drawable>();
                for (auto &entity : allEntities) {
                    auto pos = R.getComponent<ECS::Component::Position>(entity);
                    auto drawable = R.getComponent<ECS::Component::Drawable>(entity);
                    if (pos != nullptr) {
                        if (drawable->center) {
                            sf::Vector2f origin((float)drawable->sprite.getTextureRect().width / 2, (float)drawable->sprite.getTextureRect().height / 2);
                            drawable->sprite.setOrigin(origin);
                        }
                        drawable->sprite.setPosition(pos->x, pos->y);
                        _win->draw(drawable->sprite);
                    }
                }
            }
        };


        /**
        * Manages the entities's animation component and apply it to their drawable
        *
        */
        struct Animate {

            void operator()(ECS::Registry &R, double dt)
            {
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::Animation>();
                sf::Vector2u size;
                for (auto &entity : allEntities) {
                    auto anim = R.getComponent<ECS::Component::Animation>(entity);
                    auto drawable = R.getComponent<ECS::Component::Drawable>(entity);
                    if (drawable != nullptr) {
                        anim->currentFps++;
                        if (!anim->animating) {
                            anim->animating = true;
                            drawable->sprite.setTextureRect(anim->rect);
                            sf::Vector2f origin((float)anim->rect.width / 2, (float)anim->rect.height / 2);
                            drawable->sprite.setOrigin(origin);
                            continue;
                        }
                        if (anim->currentFps < anim->fps) {
                            continue;
                        }
                        anim->currentFps = 0;
                        size = drawable->sprite.getTexture()->getSize();
                        anim->rect.left += anim->rect.width;
                        if (anim->rect.left >= size.x) {
                            if (anim->destroyAtEnd) {
                                R.destroyEntity(entity);
                                continue;
                            }
                            anim->rect.left = 0;
                        }
                        drawable->sprite.setTextureRect(anim->rect);
                    }
                }
            }
        };


        /**
        * Manages the entities's parallax component and apply it to their drawable
        *
        * This cannot be used with an animation component.
        *
        */
        struct ParallaxSystem {

            void operator()(ECS::Registry &R, double dt)
            {
                // Get all entities with a Parallax component
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::Parallax>();
                sf::IntRect newRect;

                // Iterate over all entities with a Parallax component
                for (auto &entity : allEntities) {
                    auto parallax = R.getComponent<ECS::Component::Parallax>(entity);
                    auto drawable = R.getComponent<ECS::Component::Drawable>(entity);

                    if (drawable != nullptr && parallax != nullptr) {
                        newRect = drawable->sprite.getTextureRect();
                        newRect.left += parallax->speed;
                        if (newRect.left > newRect.width) {
                            newRect.left = 0;
                        }
                        drawable->sprite.setTextureRect(newRect);
                    }
                }
            }
        };

        /**
        * Debug Tool to manage local control for entities
        *
        */
        struct PlayerControl {

            std::shared_ptr<sf::RenderWindow> _win;
            sf::Event event;

            PlayerControl( std::shared_ptr<sf::RenderWindow>& winPtr) : _win(winPtr) {}

            void operator()(ECS::Registry &R, double dt)
            {
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::PlayerInput>();
                for (auto &entity : allEntities) {
                    auto inputs = R.getComponent<ECS::Component::PlayerInput>(entity);
                    inputs->action = sf::Keyboard::isKeyPressed(sf::Keyboard::Space);
                    inputs->left = sf::Keyboard::isKeyPressed(sf::Keyboard::Left);
                    inputs->up = sf::Keyboard::isKeyPressed(sf::Keyboard::Up);
                    inputs->down = sf::Keyboard::isKeyPressed(sf::Keyboard::Down);
                    inputs->right = sf::Keyboard::isKeyPressed(sf::Keyboard::Right);
                }
            }
        };

        /**
        * Debug Tool to manage local movement for entities based on Debug Movement
        *
        */
        struct PlayerMovement {

            void operator()(ECS::Registry &R, double dt)
            {
                std::vector<ECS::entity_id> allEntities = R.getEntitiesWithComponent<ECS::Component::PlayerInput>();
                for (auto &entity : allEntities) {
                    auto inputs = R.getComponent<ECS::Component::PlayerInput>(entity);
                    auto vel = R.getComponent<ECS::Component::Velocity>(entity);
                    if (vel) {
                        vel->x = (inputs->left * -300) + (inputs->right * 300);
                        vel->y = (inputs->up * -300) + (inputs->down * 300);
                    }
                }
            }
        };

    }
