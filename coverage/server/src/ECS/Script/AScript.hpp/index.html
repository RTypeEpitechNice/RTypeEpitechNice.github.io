#ifndef A_SCRIPT_H_
    #define A_SCRIPT_H_

    #include "IScript.hpp"
    #include "ECS/ECSManager.hpp"

    namespace ECS
    {
        /**
        * Default abstract of IScript, only global functions are defined.
        *
        * This script's methods will be called by the engine in specific moments, they cannot be called
        * elsewhere
        *
        * Classes that inherit from this (the scripts you will create) can have their own private variables
        * or component pointers (see example scripts in the ScriptSource folder for example)
        *
        * @param Game Pointer to access the engine class's content
        * @param _selfID ID of the script's entity
        *
        */
        class AScript : public ECS::IScript
        {
            public:
                ~AScript() = default;

                /**
                * Called on Entity's first frame of existence, usefull for getting all components once
                *
                */
                void onStart() {}

                /**
                * Called every frame
                *
                */
                void onUpdate() {}

                /**
                * Called when an entity collides with another, they need a hitbox component to be activatable
                * 
                * @param id ID of the entity who collided the script's owner
                *
                */
                void onCollision(ECS::entity_id id) {}

                /**
                * Called when mouse clicks on the entity, they need to have an hitbox
                *
                */
                void onClick() {}
                void onEnterCamera() {} //NOT SUPPORTED
                void onExitCamera() {} //NOT SUPPORTED

                /**
                * Called when Script is assigned to an entity so that they can manipulate the engine
                *
                * @param ecsManager Pointer to ECSManager Class (the engine).
                */
                void setEcsManager(ECS::ECSManager* ecsManager) {
                    Game = ecsManager;
                }

                /**
                * Defines the script's user ID
                *
                * @param id Entity ID.
                */
                void setEntityID(ECS::entity_id id) {
                    _selfID = id;
                }

                /**
                * Automatically spawn a prefab at (x, y), useful for spawning decor or temporary animation elements
                *
                * @warning NOT RECOMMENDED, MODIFYING ENTITIES CREATED BY THIS METHOD MAY CAUSE CRASH
                * @param prefabName Prefab name.
                * @param x X position
                * @param y Y position
                */
                ECS::entity_id spawnPrefabAt(std::string prefabName, int x, int y) {
                    ECS::entity_id id = Game->createEntityFromPrefab(prefabName);
                    auto e = Game->getComponent<ECS::Component::Position>(id);
                    if (e != nullptr) {
                        e->x = x;
                        e->y = y;
                    }
                    return id;
                }

            protected:
                ECS::ECSManager *Game;
                ECS::entity_id _selfID;

        };
    }

#endif
