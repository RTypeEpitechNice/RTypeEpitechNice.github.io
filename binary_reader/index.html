    binary_reader

    template<typename BasicJsonType, typename InputAdapterType, typename SAX = json_sax_dom_parser<BasicJsonType>> class binary_reader

    ////////////////
////////////////

    @brief deserialization of CBOR, MessagePack, and UBJSON values

    ia input adapter

    is_little_endian the number of characters read whether we can assume little endianness
    input_format input format
